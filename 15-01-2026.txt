#include <stdio.h>
#include <string.h>
int main() {
    char str[200];
    char *word;
    char smallest[50] = "";
    char largest[50] = "";
    fgets(str, sizeof(str), stdin);
    word = strtok(str, " \n");
    while (word != NULL) {
        if (smallest[0] == '\0' || strlen(word) < strlen(smallest)) {
            strcpy(smallest, word);
        }
        if (largest[0] == '\0' || strlen(word) > strlen(largest)) {
            strcpy(largest, word);
        }
        word = strtok(NULL, " \n");
    }
    if (smallest[0] != '\0') {
        printf("Smallest word: %s\n", smallest);
        printf("Largest word: %s\n", largest);
    } else {
        printf("No words found in the string.\n");
    }
    return 0;
}








#include <stdio.h>

int main() {
    char str[200];
    int freq[256] = {0};

    fgets(str, sizeof(str), stdin);

    for (int i = 0; str[i] != '\0'; i++) {
        freq[(unsigned char)str[i]]++;
    }

    printf("Character frequencies:\n");
    for (int i = 0; i < 256; i++) {
        if (freq[i] != 0 && i != '\n') {
            printf("'%c' occurs %d times\n", i, freq[i]);
        }
    }

    return 0;
}







#include <stdio.h>
#include<string.h>
int main() {
    char ch1[100],ch2[100];
    int count1[150]={0},count2[150]={0};
     printf("Enter first string:");
     fgets(ch1,100,stdin);
     printf("Enter second string:");
     fgets(ch2,100,stdin);
     if(strlen(ch1)!=strlen(ch2)){
        printf("Not Anagrams");
        return 0;
     }
     for(int i=0;ch1[i]!='\0';i++){
        count1[(int)ch1[i]]++;
         count2[(int)ch2[i]]++;
     }
     for(int i=0;i<150;i++){
        if(count1[i]!=count2[i]){
            printf("Not Anagrams");
            return 0;
        }
     }
     printf("Anagrams");
    return 0;
}











#include <stdio.h>
#include<string.h>
int main() {
    char ch[100],temp;
    int i,j,len;
    printf("Enter a string :");
    fgets(ch,100,stdin);
    len = strlen(ch);
    for(i=0;i<len-1;i++){
        for(j=i+1;j<len;j++){
            if(ch[i]>ch[j]){
                temp = ch[i];
                ch[i] = ch[j];
                ch[j] = temp;
            }
        }
    }
    printf("sorted string:%s",ch);
    return 0;
}
