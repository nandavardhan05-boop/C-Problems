 #include <stdio.h>
#include <string.h>

int main() {
    char code[50];
    int i, length, flag = 1;
    scanf("%s", code);
    length = strlen(code);
    for (i = 0; i < length / 2; i++) {
        if (code[i] != code[length - i - 1]) {
            flag = 0;
            break;
        }
    }
    if (flag == 1)
        printf("Palindrome");
    else
        printf("Not Palindrome");
    return 0;
}
