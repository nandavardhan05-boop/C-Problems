 #include<stdio.h>
int main()
{
    int size,count =0;
    scanf("%d",&size);
    int arr[size];
    for(int i=0;i<size;i++){
        scanf("%d",&arr[i]);
    }
    for(int i=0;i<size;i++){
        if(arr[i]>0){
            count++;
        }
    }
   printf("%d",count);
   return 0;
}
