#include <stdio.h>
#include <string.h>

int main() {
    char url[100];
    fgets(url, sizeof(url), stdin);
    url[strcspn(url, "\n")] = '\0';
    if (strncmp(url, "http://", 7) == 0 || strncmp(url, "https://", 8) == 0) {
        printf("Valid");
    } else {
        printf("Invalid");
    }
    return 0;
}
