 #include <stdio.h>

int main() {
    char str[200];
    int i, words = 0;

    printf("Enter a sentence:\n");
    gets(str);   

    for (i = 0; str[i] != '\0'; i++) {
        if ((i == 0 && str[i] != ' ') || 
            (str[i] != ' ' && str[i - 1] == ' ')) {
            words++;
        }
    }

    printf("Number of words = %d\n", words);

    return 0;
}
   

#include <stdio.h>
#include <string.h>

int main() {
    char code[100];
    int i, len, flag = 1;

    printf("Enter product code: ");
    scanf("%s", code);   

    len = strlen(code);

    for (i = 0; i < len / 2; i++) {
        if (code[i] != code[len - i - 1]) {
            flag = 0;
            break;
        }
    }

    if (flag)
        printf("The product code is a Palindrome\n");
    else
        printf("The product code is NOT a Palindrome\n");

    return 0;
}




#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main() {
    char otp[20];
    int i, length, valid = 1;

    printf("Enter the 6-digit OTP: ");
    scanf("%s", otp);

    length = strlen(otp);
    if (length != 6) {
        valid = 0;
    } else {
        for (i = 0; i < length; i++) {
            if (!isdigit(otp[i])) {
                valid = 0;
                break;
            }
        }
    }

    if (valid)
        printf("Valid OTP. Authentication successful.\n");
    else
        printf("Invalid OTP. Please enter a 6-digit numeric OTP.\n");

    return 0;
}






#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main() {
    char name[100];
    int i;
    printf("Enter the customer name: ");
    scanf("%s", name);
    name[0] = toupper(name[0]);
    for (i = 1; name[i] != '\0'; i++) {
        name[i] = tolower(name[i]);
    }

    printf("Formatted Name: %s\n", name);

    return 0;
}
