#include<stdio.h>
int main()
{
    int n;
    scanf("%d",&n);
    int rev=0,remainder;
    while( n!=0){
        remainder=n%10;
        rev=rev*10+remainder;
        n=n/10;
        
    }
   printf( "%d",rev);
    return 0;
}
